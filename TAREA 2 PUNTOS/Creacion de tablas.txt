
CREATE TABLE GUARDIA (
	GUAR_CEDULA varchar(10) NOT NULL,
	GUAR_NOMBRE varchar(50),
	GUAR_APELLIDO varchar(50),
	GUAR_FECH_NACIMIENTO date,
	GUAR_TIPO_SANGRE varchar(10),
	GUAR_CELULAR varchar(10),
	GUAR_CORREO varchar(50),
	CONSTRAINT GUARDIA_pk PRIMARY KEY (GUAR_CEDULA)

);

CREATE TABLE EQUIPO (
	EQUI_IDinteger NOT NULL,
	EQUI_TALLA varchar(20),
	EQUI_PESO varchar(10),
	EQUI_FECHA date,
	EQUI_DETALLE varchar(60),
	GUAR_CEDULA_GUARDIA varchar(10) NOT NULL,
	CONSTRAINT EQUIPO_pk PRIMARY KEY (EQUI_ID)
);

CREATE TABLE ENTRENAMIENTO (
	ENTRE_ID integer NOT NULL,
	ENTRE_TIPO varchar(50),
	ENTRE_FECHA_INICIO date,
	ENTRE_FECHA_FINAL date,
	ENTRE_DURACION varchar(20),
	ENTRE_ESTADO varchar(20),
	ENTRE_COSTO float,
	CONSTRAINT ENTRENAMIENTO_pk PRIMARY KEY (ENTRE_ID)

);

CREATE TABLE ASCENSO (
	ASCEN_ID integer NOT NULL,
	ASCEN_FECHA timestamp,
	ASCEN_CARGO varchar(50),
	ASCEN_DETALLE varchar(50),
	GUAR_CEDULA_GUARDIA varchar(10) NOT NULL,
	CONSTRAINT ASCENSO_pk PRIMARY KEY (ASCEN_ID)

);
 
CREATE TABLE EMPRESA (
	EMPRE_RUC varchar(20) NOT NULL,
	EMPRE_NOMBRE varchar(50),
	EMPRE_DIRECCION varchar(50),
	EMPRE_TELEFONO varchar(10),
	EMPRE_CORREO varchar(50),
	EMPRE_PAGINA_WEB varchar(50),
	CONSTRAINT EMPRESA_pk PRIMARY KEY (EMPRE_RUC)

);
 
CREATE TABLE CONVENIO (
	CONVE_NUMERO integer NOT NULL,
	CONVE_TIPO varchar(50),
	CONVE_DETALLE varchar(50),
	CONVE_DURACION varchar(40),
	CONVE_FECHA timestamp,
	CONVE_PAGO_MENSUAL float,
	EMPRE_RUC_EMPRESA varchar(20) NOT NULL,
	GUAR_CEDULA_GUARDIA varchar(10) NOT NULL,
	CONSTRAINT CONVENIOS_pk PRIMARY KEY (CONVE_NUMERO)

);
 
CREATE TABLE HORARIO (
	HORA_ID integer NOT NULL,
	HORA_FECHA_INICIO date,
	HORA_FECHA_FINAL date,
	HORA_TOTAL_DIAS_LABORABLES varchar(30),
	CONVE_NUMERO_CONVENIO integer NOT NULL,
	CONSTRAINT HORARIO_pk PRIMARY KEY (HORA_ID)
);
 
CREATE TABLE INCONVENENTE (
	INCON_ID varchar(10) NOT NULL,
	INCON_FECHA date,
	INCON_DETALLE varchar(70),
	GUAR_CEDULA_GUARDIA varchar(10) NOT NULL,
	EMPRE_RUC_EMPRESA varchar(20) NOT NULL,
	CONSTRAINT INCONVENENTES_pk PRIMARY KEY (INCON_ID)

);
 
CREATE TABLE SERVICIO (
	SERV_ID varchar(20) NOT NULL,
	SERV_DETALLE varchar(60),
	CONSTRAINT SERVICIO_pk PRIMARY KEY (SERV_ID)
);

CREATE TABLE FACTURACION (
	FACT_NUM varchar(10) NOT NULL,
	FACT_FECHA timestamp,
	FACT_TOTAL float,
	FACT_SUBTOTAL float,
	GUAR_CEDULA_GUARDIA varchar(10) NOT NULL,
	EMPRE_RUC_EMPRESA varchar(20) NOT NULL,
	CONSTRAINT FACTURACION_pk PRIMARY KEY (FACT_NUM)
);
 
CREATE TABLE DETALLE_FACTURA (
	DETA_ID smallint NOT NULL,
	DETA_CANTIDAD integer,
	DETA_PRECIO float,
	DETA_TOTAL float,
	FACT_NUM_FACTURACION varchar(10) NOT NULL,
	CONVE_NUMERO_CONVENIO integer NOT NULL,
	CONSTRAINT DETALLE_FACTURA_pk PRIMARY KEY (DETA_ID)
);

CREATE TABLE GUARDIA_ENTRENAMIENTO (
	GUAR_ENTRE_CODIGO varchar(20) NOT NULL,
	GUAR_CEDULA_GUARDIA varchar(10) NOT NULL,
	ENTRE_ID_ENTRENAMIENTO integer NOT NULL,
	CONSTRAINT GUARDIA_ENTRENAMIENTO_pk PRIMARY KEY (GUAR_ENTRE_CODIGO)
);

CREATE TABLE GUARDIA_SERVICIO (
	SERV_GUAR_ID varchar(5) NOT NULL,
	SERV_GUAR_FECHA date,
	SERV_GUAR_DETALLE varchar(50),
	SERV_ID_SERVICIO varchar(20) NOT NULL,
	GUAR_CEDULA_GUARDIA varchar(10) NOT NULL,
	CONSTRAINT GUARDIA_SERVICIO_pk PRIMARY KEY (SERV_GUAR_ID)
);
